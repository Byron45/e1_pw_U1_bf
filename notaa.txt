package com.uce.ec.clientes.rest;

import com.uce.ec.clientes.clients.PrestamoRestClient;
import com.uce.ec.clientes.dtos.ClienteResumenDto;
import com.uce.ec.clientes.repo.ClienteRepository;
import jakarta.enterprise.context.ApplicationScoped;
import jakarta.inject.Inject;
import jakarta.transaction.Transactional;
import jakarta.ws.rs.*;
import jakarta.ws.rs.core.MediaType;
import jakarta.ws.rs.core.Response;
import org.eclipse.microprofile.rest.client.inject.RestClient;

@Path("/clientes")
@Produces(MediaType.APPLICATION_JSON)
@Consumes(MediaType.APPLICATION_JSON)
@Transactional
@ApplicationScoped
public class ClienteRest {

    final ClienteRepository clienteRepository;
    final PrestamoRestClient prestamosClient;

    @Inject
    public ClienteRest(ClienteRepository clienteRepository,
                       @RestClient PrestamoRestClient prestamosClient) {
        this.clienteRepository = clienteRepository;
        this.prestamosClient = prestamosClient;
    }

    @GET
    @Path("/{cedula}/prestamos")
    public Response findPrestamosByCedula(@PathParam("cedula") String cedula) {
        return clienteRepository.findByCedula(cedula)
                .map(cliente -> {
                    var prestamos = prestamosClient.findByCliente(cliente.getId());

                    double montoTotal = prestamos.stream()
                            .mapToDouble(p -> p.getMonto_total() != null ? p.getMonto_total() : 0.0)
                            .sum();

                    return ClienteResumenDto.builder()
                            .cedula(cliente.getCedula())
                            .email(cliente.getEmail())
                            .montoTotalPrestamos(montoTotal)
                            .prestamos(prestamos)
                            .build();
                })
                .map(Response::ok)
                .orElse(Response.status(Response.Status.NOT_FOUND))
                .build();
    }
}
